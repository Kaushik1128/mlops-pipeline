# --- Core data + numerics ---
pandas==2.2.3
numpy==2.1.3
pyarrow==18.1.0
# pyarrow gives us Parquet read/write. Parquet is ~5x smaller than CSV and
# preserves dtypes — better than CSV for any non-trivial ML pipeline.

# --- Modeling ---
scikit-learn==1.6.0
imbalanced-learn==0.13.0
# imbalanced-learn for SMOTE + other resampling. We'll need it in Phase 3
# because 0.17% fraud means a default classifier will trivially "predict
# zero" and score 99.83% accuracy while detecting no fraud.

xgboost==3.3.0
# Gradient-boosted trees — our champion model for tabular fraud data. Has a
# built-in `scale_pos_weight` parameter designed for class imbalance, which
# we tune to our ~577:1 ratio in Phase 3.
# NOTE: must be >=2.1.4 — earlier versions lack sklearn 1.6's __sklearn_tags__
# API and crash on predict_proba when wrapped in an sklearn Pipeline.

# --- Experiment tracking client ---
mlflow==2.18.0
# Same version as the server in docker/mlflow/. The client just makes HTTP
# calls to http://localhost:5000 — Python version on each side is independent.

# --- Drift detection / monitoring ---
evidently==0.7.21
# Compares a reference vs current dataset with per-feature statistical tests
# (normalized Wasserstein distance here) and flags dataset-level drift. The
# drift signal it emits is the trigger for Phase 6 auto-retraining.

# --- Model serving ---
bentoml==1.3.22
# Serving framework — wraps the trained model in a high-performance REST API
# with typed request validation, and can build a deployable container image.
# We import the @staging model from the MLflow registry into BentoML's store.

setuptools<81
# BentoML's MLflow integration still imports pkg_resources, which setuptools
# removed in v81. Pin below 81 so pkg_resources remains available.

# --- Data versioning ---
dvc[s3]==3.58.0
# dvc[s3] = DVC + the s3 extras (boto3 driver). MinIO speaks S3, so this is
# what we need to push/pull data through MinIO.

pathspec==0.12.1
# DVC 3.58 imports an internal symbol (_DIR_MARK) from pathspec that was
# removed in pathspec 1.x. Pin to the 0.12 line until DVC catches up.
# Without this, `dvc init` errors with "cannot import name '_DIR_MARK'".

# --- Visualization (notebooks) ---
matplotlib==3.10.0
seaborn==0.13.2

# --- Notebook environment ---
jupyterlab==4.3.4
ipywidgets==8.1.5

# --- Misc utilities ---
python-dotenv==1.0.1
# python-dotenv lets our Python scripts read the .env file (same one Compose
# uses) without us having to manually set OS-level env vars.

# --- Modeling ---

imbalanced-learn==0.13.0
# imbalanced-learn for SMOTE + other resampling. We'll need it in Phase 3

# --- Experiment tracking client ---
mlflow==2.18.0
# Same version as the server in docker/mlflow/. The client just makes HTTP
# calls to http://localhost:5000 — Python version on each side is independent.

# --- Data versioning ---
dvc[s3]==3.58.0
# dvc[s3] = DVC + the s3 extras (boto3 driver). MinIO speaks S3, so this is
# what we need to push/pull data through MinIO.

pathspec==0.12.1
# DVC 3.58 imports an internal symbol (_DIR_MARK) from pathspec that was
# removed in pathspec 1.x. Pin to the 0.12 line until DVC catches up.
# Without this, `dvc init` errors with "cannot import name '_DIR_MARK'".
